{{/* @see - https://gohugo.io/render-hooks/images/ available values: ![Description](Destination "Title") - .Attributes (map) the Markdown attributes. - .Destination (cast.ToString) The image destination. - .IsBlock (bool) Returns true if a standalone image is not wrapped within a paragraph element. - .Ordinal (cast.ToInt) The zero-based ordinal of the image on the page. - .Page (page) A reference to the page containing the image. - .PlainText (cast.ToString) The image description as plain text. - .Text (cast.ToString) The image description. - .Title (cast.ToString) The image title. */}} {{- $context := .PageInner -}} {{- $label := .Text -}} {{- $src := .Destination -}} {{- $isPageBundle := true -}} {{- $isStaticFile := false -}} {{- $config := partials.IncludeCached "func/pictures/getConfiguration.html" . -}} {{- $default_format := $config.default_format | compare.Default "webp" -}} {{/* retrieve the article image */}} {{- $options := dict "path" $src "page" $context.RelPermalink -}} {{- $localimage := partials.Include "func/pictures/getImageResource.html" $options $options -}} {{- if compare.Eq false $localimage -}} {{- $imageOptions := dict "image" $src "title" $label -}} {{- partials.Include "images/markup-image-simple.html" $imageOptions -}} {{- else -}} {{/* setup */}} {{- $link := false -}} {{- $aspect_ratio := partials.IncludeCached "func/pictures/getAspectRatio.html" . -}} {{- $image_sizes := partials.IncludeCached "func/pictures/getImageSizes.html" . -}} {{- $break_points := partials.IncludeCached "func/pictures/getBreakPoints.html" . -}} {{/* preparing aspect_ratio image sizes - once sitewide */}} {{- $arOptions := dict "image_sizes" $image_sizes "aspect_ratio" $aspect_ratio -}} {{- $image_sizes2 := partials.IncludeCached "func/pictures/getAspectRatioImagesizes.html" $arOptions $arOptions -}} {{- with $localimage -}} {{- $image := . -}} {{/* evaluate, if the image is collections.In a format that can be processed */}} {{- $opt := dict "isPageBundle" $isPageBundle "image" $image -}} {{- $isProcessable := partials.IncludeCached "func/pictures/isProcessable.html" $opt $opt -}} {{- if compare.Eq "false" $isProcessable -}} {{- $isPageBundle := false -}} {{- $isStaticFile := true -}} {{- end -}} {{- $title := "" -}} {{/* if a strings.Title is set via frontmatter then use it as caption */}} {{- if compare.Ne $image.Title $image.Name -}} {{- $title = $image.Title | transform.Markdownify -}} {{- end -}} {{- $images := collections.Slice -}} {{- if $isStaticFile -}} {{/* no image processing for non-raster images and animated gifs */}} {{- $images = $images | collections.Append $localimage -}} {{- else -}} {{- range $index, $value := $image_sizes -}} {{- if compare.Ge $localimage.Width $value -}} {{- $imagesize := fmt.Printf "%sx %s" (cast.ToString $value) $default_format -}} {{- $images = $images | collections.Append ($image.Resize $imagesize) -}} {{- end -}} {{- end -}} {{- end -}} {{- $imageOptions := dict "images" $images "link" $link "break_points" $break_points "title" $title "label" $label "context" $context -}} {{- partials.Include "images/markup-image.html" $imageOptions -}} {{- end -}} {{- end -}}

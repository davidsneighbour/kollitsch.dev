{{- /* Retrieves the value of `.Inner` or false if nothing is retrieved. This function is trimming spaces and whitespaces from the beginning and end of the value. Usage: ```gohtml {{- with (partial "func/getInner.html" .) -}} The dot is the retrieved .Inner value. {{- end -}} ``` */ -}} {{- $inner := (strings.Chomp .Inner) -}} {{- $inner = trim $inner " " -}} {{- if eq $inner "" -}} {{- $inner = false -}} {{- end -}} {{- return $inner -}}
